# Phase 1 — Chat Foundation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a working role-aware chat across two surfaces (REPL terminal + web app) backed by a persistent FastAPI service. Demoable in 5 minutes; foundation for every later phase.

**Architecture:** FastAPI async backend with SSE streaming `/api/v1/chat`. Postgres persists `users`, `chat_sessions`, `chat_messages`. Two clients consume the same backend: a Python REPL (`scripts/repl_chat.py`) and a Vite/React UI (`frontend/src/features/chat/`). System prompt switches on `users.role` ∈ {client, advisor}. OpenAI calls go through a thin async wrapper with a deterministic stub fallback when no API key is set, so tests run offline.

**Tech Stack:** FastAPI · SQLAlchemy 2 async · Alembic · asyncpg · OpenAI Python SDK ≥ 1.54 · Vite · React 19 · TypeScript · Tailwind · shadcn/ui · pytest · Vitest.

**Spec coverage:** C1, C2, AC2, AC3 (partial — covers per-user history). C4 (system diagram) is covered by `docs/implementation-plan.md` §2.

```python
# backend/app/services/llm.py
"""Async LLM wrapper. Falls back to a deterministic stub if no API key.

The stub keeps tests offline and the REPL working without billing.
"""
import logging
from collections.abc import AsyncIterator

from openai import AsyncOpenAI

from app.core.config import settings

log = logging.getLogger(__name__)
_stub_warned = False


def _client() -> AsyncOpenAI:
    return AsyncOpenAI(api_key=settings.openai_api_key)


async def _stub_stream(messages: list[dict]) -> AsyncIterator[str]:
    last_user = next(
        (m["content"] for m in reversed(messages) if m["role"] == "user"), ""
    )
    text = f"[stub] echo: {last_user}"
    for ch in text:
        yield ch


async def stream_chat(
    messages: list[dict], model: str | None = None
) -> AsyncIterator[str]:
    """Yield content deltas from the LLM (or stub)."""
    global _stub_warned
    if not settings.openai_api_key:
        if not _stub_warned:
            log.warning("OPENAI_API_KEY not set; using deterministic stub stream")
            _stub_warned = True
        async for tok in _stub_stream(messages):
            yield tok
        return

    client = _client()
    stream = await client.chat.completions.create(
        model=model or settings.openai_model,
        messages=messages,
        stream=True,
    )
    async for chunk in stream:
        delta = chunk.choices[0].delta.content
        if delta:
            yield delta
```

```python
# backend/app/routers/chat.py
"""Chat router: users, sessions, messages, SSE streaming."""
import uuid

from fastapi import APIRouter, Depends, HTTPException, Response, status
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.deps import get_session
from app.db.models import ChatMessage, ChatSession, MessageRole, User, UserRole
from app.schemas.chat import (
    MessageOut,
    SessionCreate,
    SessionOut,
    UserCreate,
    UserOut,
)

router = APIRouter(prefix="/api/v1", tags=["chat"])


@router.post("/users", response_model=UserOut)
async def upsert_user(
    body: UserCreate,
    response: Response,
    db: AsyncSession = Depends(get_session),
) -> User:
    existing = (
        await db.execute(select(User).where(User.email == body.email))
    ).scalar_one_or_none()
    if existing:
        response.status_code = status.HTTP_200_OK
        return existing
    user = User(
        email=body.email,
        display_name=body.display_name,
        role=UserRole(body.role),
    )
    db.add(user)
    await db.commit()
    await db.refresh(user)
    response.status_code = status.HTTP_201_CREATED
    return user


@router.post("/sessions", response_model=SessionOut, status_code=201)
async def create_session_(
    body: SessionCreate,
    db: AsyncSession = Depends(get_session),
) -> ChatSession:
    user = await db.get(User, body.user_id)
    if not user:
        raise HTTPException(404, "user not found")
    sess = ChatSession(user_id=body.user_id, title=body.title)
    db.add(sess)
    await db.commit()
    await db.refresh(sess)
    return sess


@router.get("/sessions", response_model=list[SessionOut])
async def list_sessions(
    user_id: uuid.UUID,
    db: AsyncSession = Depends(get_session),
) -> list[ChatSession]:
    rows = (
        await db.execute(
            select(ChatSession)
            .where(ChatSession.user_id == user_id)
            .order_by(ChatSession.created_at.desc())
        )
    ).scalars().all()
    return list(rows)


@router.get("/sessions/{session_id}/messages", response_model=list[MessageOut])
async def list_messages(
    session_id: uuid.UUID,
    db: AsyncSession = Depends(get_session),
) -> list[ChatMessage]:
    rows = (
        await db.execute(
            select(ChatMessage)
            .where(ChatMessage.session_id == session_id)
            .order_by(ChatMessage.created_at.asc())
        )
    ).scalars().all()
    return list(rows)
```

```python
async def chat_stream(
    body: ChatRequest,
    db: AsyncSession = Depends(get_session),
) -> StreamingResponse:
    sess = await db.get(ChatSession, body.session_id)
    if not sess:
        raise HTTPException(404, "session not found")
    user = await db.get(User, sess.user_id)
    assert user is not None

    # Persist user turn first so reload mid-stream still shows it.
    user_msg = ChatMessage(
        session_id=sess.id, role=MessageRole.user, content=body.content
    )
    db.add(user_msg)
    await db.commit()

    history = (
        await db.execute(
            select(ChatMessage)
            .where(ChatMessage.session_id == sess.id)
            .order_by(ChatMessage.created_at.asc())
        )
    ).scalars().all()

    sysprompt = CLIENT_SYSTEM if user.role is UserRole.client else ADVISOR_SYSTEM
    llm_messages = [{"role": "system", "content": sysprompt}]
    llm_messages += [{"role": m.role.value, "content": m.content} for m in history]

    async def event_gen():
        chunks: list[str] = []
        async for tok in stream_chat(llm_messages):
            chunks.append(tok)
            yield f"data: {json.dumps({'delta': tok})}\n\n"

            await s2.refresh(asst)
            yield f"data: {json.dumps({'done': True, 'message_id': str(asst.id)})}\n\n"
```

```python
Talks to the DB directly — no FastAPI server required.
Usage:

import sys
import uuid

from sqlalchemy import select

from app.db.models import ChatMessage, ChatSession, MessageRole, User, UserRole
from app.db.session import AsyncSessionLocal
from app.services.llm import stream_chat

ADVISOR_SYSTEM = (
    "You are AuraWealth's command-center copilot for a wealth advisor."
)


async def _upsert_user(email: str, name: str, role: UserRole) -> User:
    async with AsyncSessionLocal() as s:
        existing = (
            await s.execute(select(User).where(User.email == email))
        ).scalar_one_or_none()
        if existing:
            return existing
        u = User(email=email, display_name=name, role=role)
        s.add(u)
        await s.commit()
        await s.refresh(u)
        return u


async def _get_or_create_session(user_id: uuid.UUID, sid: uuid.UUID | None) -> ChatSession:
    async with AsyncSessionLocal() as s:
        if sid:
            sess = await s.get(ChatSession, sid)
            if sess:
                return sess
        sess = ChatSession(user_id=user_id, title="REPL")
        s.add(sess)
        await s.commit()
        await s.refresh(sess)
        return sess


async def _history(session_id: uuid.UUID) -> list[ChatMessage]:
    async with AsyncSessionLocal() as s:
        rows = (
            await s.execute(
                select(ChatMessage)
                .where(ChatMessage.session_id == session_id)
                .order_by(ChatMessage.created_at.asc())
            )
        ).scalars().all()
        return list(rows)


async def _persist(session_id: uuid.UUID, role: MessageRole, content: str) -> None:
    async with AsyncSessionLocal() as s:
        s.add(ChatMessage(session_id=session_id, role=role, content=content))
        await s.commit()


async def _run(args: argparse.Namespace) -> int:
    role = UserRole(args.role)
    name = args.name or args.email.split("@", 1)[0]
    user = await _upsert_user(args.email, name, role)
    sess = await _get_or_create_session(user.id, args.session_id)

    sysprompt = CLIENT_SYSTEM if role is UserRole.client else ADVISOR_SYSTEM

    print(f"AuraWealth REPL — user={user.email} role={role.value} session={sess.id}")
    print("Type /exit, /quit, or Ctrl-D to leave. /history to dump turns.")
    print()

    while True:
        try:
            line = input("> ").strip()
        except EOFError:
            print()
            return 0
        if not line:
            continue
        if line in {"/exit", "/quit"}:
            return 0
        if line == "/history":
            for m in await _history(sess.id):
                print(f"  [{m.role.value}] {m.content}")
            continue

        await _persist(sess.id, MessageRole.user, line)
        history = await _history(sess.id)
        msgs = [{"role": "system", "content": sysprompt}]
        msgs += [{"role": m.role.value, "content": m.content} for m in history]

        chunks: list[str] = []
        async for tok in stream_chat(msgs):
            sys.stdout.write(tok)
            sys.stdout.flush()
            chunks.append(tok)
        sys.stdout.write("\n")
        await _persist(sess.id, MessageRole.assistant, "".join(chunks))
```

```ts
// frontend/src/lib/chat-api.ts
const BASE = (import.meta.env.VITE_API_BASE_URL as string | undefined) ?? '/api/v1'

export type Role = 'client' | 'advisor'
export interface User { id: string; email: string; display_name: string; role: Role }
export interface Session { id: string; user_id: string; title: string | null; created_at: string; updated_at: string }
export interface Message { id: string; session_id: string; role: 'user' | 'assistant' | 'system'; content: string; created_at: string }

async function jpost<T>(path: string, body: unknown): Promise<T> {
  const r = await fetch(`${BASE}${path}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  })
  if (!r.ok) throw new Error(`${path} → ${r.status}`)
  return r.json() as Promise<T>
}

async function jget<T>(path: string): Promise<T> {
  const r = await fetch(`${BASE}${path}`)
  if (!r.ok) throw new Error(`${path} → ${r.status}`)
  return r.json() as Promise<T>
}

export const upsertUser = (input: { email: string; display_name: string; role: Role }) =>
  jpost<User>('/users', input)

export const createSession = (user_id: string, title?: string) =>
  jpost<Session>('/sessions', { user_id, title })

export const listSessions = (user_id: string) =>
  jget<Session[]>(`/sessions?user_id=${encodeURIComponent(user_id)}`)

export const listMessages = (session_id: string) =>
  jget<Message[]>(`/sessions/${session_id}/messages`)

export interface StreamHandlers {
  onDelta(t: string): void
  onDone(messageId: string): void
  onError(err: Error): void
}

export async function streamChat(
  session_id: string,
  content: string,
  h: StreamHandlers,
  signal?: AbortSignal,
): Promise<void> {
  const r = await fetch(`${BASE}/chat`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ session_id, content }),
    signal,
  })
  if (!r.ok || !r.body) {
    h.onError(new Error(`/chat → ${r.status}`))
    return
  }
  const reader = r.body.getReader()
  const dec = new TextDecoder()
  let buf = ''
  for (;;) {
    const { value, done } = await reader.read()
    if (done) break
    buf += dec.decode(value, { stream: true })
    const events = buf.split('\n\n')
    buf = events.pop() ?? ''
    for (const ev of events) {
      const line = ev.split('\n').find(l => l.startsWith('data: '))
      if (!line) continue
      try {
        const obj = JSON.parse(line.slice(6))
        if ('delta' in obj) h.onDelta(obj.delta)
        else if (obj.done) h.onDone(obj.message_id)
      } catch (e) {
        h.onError(e as Error)
      }
    }
  }
}
```

```ts
import { useCallback, useEffect, useRef, useState } from 'react'
import {
  createSession, listSessions, listMessages, streamChat, upsertUser,
  type Message, type Role, type Session, type User,
} from '@/lib/chat-api'

const SEED: Record<Role, { email: string; display_name: string }> = {
  client: { email: 'client-demo@aura.test', display_name: 'Demo Client' },
  advisor: { email: 'advisor-demo@aura.test', display_name: 'Demo Advisor' },
}

const LS_ROLE = 'aura.role'
const LS_USER = (role: Role) => `aura.user.${role}`
const LS_SESS = (role: Role) => `aura.session.${role}`

export function useChat() {
  const [role, setRoleState] = useState<Role>(() =>
    (localStorage.getItem(LS_ROLE) as Role | null) ?? 'client'
  )
  const [user, setUser] = useState<User | null>(null)
  const [session, setSession] = useState<Session | null>(null)
  const [messages, setMessages] = useState<Message[]>([])
  const [isStreaming, setIsStreaming] = useState(false)
  const [error, setError] = useState<Error | null>(null)
  const abortRef = useRef<AbortController | null>(null)

  // Bootstrap user + session on role change
  useEffect(() => {
    let cancelled = false
    ;(async () => {
      try {
        const seed = SEED[role]
        const u = await upsertUser({ ...seed, role })
        if (cancelled) return
        localStorage.setItem(LS_USER(role), u.id)

        const cachedSid = localStorage.getItem(LS_SESS(role))
        let s: Session | null = null
        if (cachedSid) {
          const list = await listSessions(u.id)
          s = list.find(x => x.id === cachedSid) ?? null
        }
        if (!s) {
          const all = await listSessions(u.id)
          s = all[0] ?? (await createSession(u.id))
        }
        if (cancelled) return
        localStorage.setItem(LS_SESS(role), s.id)

        const msgs = await listMessages(s.id)
        if (cancelled) return
        setUser(u); setSession(s); setMessages(msgs); setError(null)
      } catch (e) {
        if (!cancelled) setError(e as Error)
      }
    })()
    return () => { cancelled = true }
  }, [role])

  const setRole = useCallback((r: Role) => {
    abortRef.current?.abort()
    localStorage.setItem(LS_ROLE, r)
    setUser(null); setSession(null); setMessages([])
    setRoleState(r)
  }, [])

  const send = useCallback(async (content: string) => {
    if (!session || !content.trim() || isStreaming) return
    setError(null)
    setIsStreaming(true)
    const tempUser: Message = {
      id: `tmp-u-${Date.now()}`, session_id: session.id,
      role: 'user', content, created_at: new Date().toISOString(),
    }
    const tempAsst: Message = {
      id: `tmp-a-${Date.now()}`, session_id: session.id,
      role: 'assistant', content: '', created_at: new Date().toISOString(),
    }
    setMessages(prev => [...prev, tempUser, tempAsst])

    const ac = new AbortController()
    abortRef.current = ac
    try {
      await streamChat(session.id, content, {
        onDelta: tok => setMessages(prev => {
          const next = [...prev]
          const last = next[next.length - 1]
          next[next.length - 1] = { ...last, content: last.content + tok }
          return next
        }),
        onDone: id => setMessages(prev => {
          const next = [...prev]
          const last = next[next.length - 1]
          next[next.length - 1] = { ...last, id }
          return next
        }),
        onError: e => setError(e),
      }, ac.signal)
    } finally {
      setIsStreaming(false)
    }
  }, [session, isStreaming])

  return { user, role, setRole, session, messages, send, isStreaming, error }
}
```

```tsx
  onSend, disabled,
}: { onSend: (text: string) => void; disabled: boolean }) {
  const [value, setValue] = useState('')
  const submit = () => {
    const v = value.trim()
    if (!v || disabled) return
    onSend(v)
    setValue('')
  }
  const onKeyDown = (e: KeyboardEvent<HTMLTextAreaElement>) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault()
      submit()
    }
  }
  return (
    <div className="flex items-end gap-2 border-t p-3">
      <Textarea
        value={value}
        onChange={e => setValue(e.target.value)}
        onKeyDown={onKeyDown}
        placeholder="Ask anything…"
        disabled={disabled}
        rows={2}
      />
      <Button onClick={submit} disabled={disabled || !value.trim()}>Send</Button>
    </div>
  )
}
```
